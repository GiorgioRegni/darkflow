Pour les impatients

~ $ su -c "adduser $(id -un) sudo"

# Logout puis login si nécessaire

~ $ curl -s http://darkflow.org/live.sh | bash

# Mettez vos images dans ~/DARKFLOW/ et commencez le traitement !
